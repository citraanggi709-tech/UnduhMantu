${namaAddon} Rp ${hargaAddon.toLocaleString('id-ID')}
`; } }); // Tampilkan atau sembunyikan section addon di invoice jika tidak diceklis let addonContainer = document.getElementById('invoiceAddonContainer'); if(hasAddon) { addonContainer.style.display = 'block'; document.getElementById('addonList').innerHTML = addonHtml; } else { addonContainer.style.display = 'none'; } let totalSemua = hargaPaket + totalAddon; document.getElementById('displayTotal').innerText = 'Rp ' + totalSemua.toLocaleString('id-ID'); } function prosesBayar() { let namaPria = document.getElementById('pria').value.trim(); let namaWanita = document.getElementById('wanita').value.trim(); let noWa = document.getElementById('whatsapp').value.trim(); let totalBayar = document.getElementById('displayTotal').innerText; // Mendapatkan metode pembayaran yang dipilih pelanggan let metodeBayar = "Transfer Bank"; if(document.getElementById('payBca').checked) metodeBayar = "Transfer BCA"; if(document.getElementById('payMandiri').checked) metodeBayar = "Bank Mandiri"; if(document.getElementById('payQris').checked) metodeBayar = "QRIS Instan"; if(namaPria === "" || namaWanita === "" || noWa === "") { alert("Mohon lengkapi data pasangan pengantin dan nomor WhatsApp Anda terlebih dahulu."); return; } // Teks otomatis yang akan masuk ke WhatsApp Anda let teksWa = `Halo NikahDigital, saya ingin mengonfirmasi pesanan Undangan Pernikahan.\n\n` + `🤵 Mempelai Pria: ${namaPria}\n` + `👰 Mempelai Wanita: ${namaWanita}\n` + `📱 No. WhatsApp Klien: +62${noWa}\n` + `💳 Metode Pembayaran: ${metodeBayar}\n` + `💰 *Total Pembayaran:* ${totalBayar}\n\n` + `Mohon kirimkan detail petunjuk pembayaran/proses selanjutnya.
